-- 1. TABELA DE USU√ÅRIOS (PERFIS)
create table if not exists profiles (
  id uuid primary key default gen_random_uuid(),
  username text unique not null,
  password text not null,
  name text not null,
  avatar text,
  role text,
  status text default 'ATIVO',
  is_admin boolean default false,
  created_at timestamp with time zone default now()
);

-- 2. TABELA DE PROJETOS
create table if not exists projects (
  id uuid primary key default gen_random_uuid(),
  title text not null,
  description text,
  responsible_id uuid references profiles(id),
  assigned_user_ids uuid[] default '{}',
  status text default 'BACKLOG',
  address text,
  neighborhood text,
  created_at timestamp with time zone default now(),
  updated_at timestamp with time zone default now()
);

-- 3. TABELA DE TAREFAS (CHECKLIST COM RIs)
create table if not exists tasks (
  id uuid primary key default gen_random_uuid(),
  project_id uuid references projects(id) on delete cascade,
  title text not null,
  completed boolean default false,
  stage text,
  responsible_id uuid references profiles(id),
  observations text default '', -- Coluna para observa√ß√µes t√©cnicas
  completed_at timestamp with time zone,
  created_at timestamp with time zone default now()
);

-- 4. TABELA DE COMUNICA√á√ÉO (MENSAGENS/COMENT√ÅRIOS)
create table if not exists comments (
  id uuid primary key default gen_random_uuid(),
  project_id uuid references projects(id) on delete cascade,
  author_id uuid references profiles(id),
  author_name text,
  content text not null,
  created_at timestamp with time zone default now()
);

-- 5. HABILITAR REALTIME
begin;
  drop publication if exists supabase_realtime;
  create publication supabase_realtime;
commit;
alter publication supabase_realtime add table projects, tasks, comments;

-- 6. DESABILITAR RLS PARA DESENVOLVIMENTO
alter table profiles disable row level security;
alter table projects disable row level security;
alter table tasks disable row level security;
alter table comments disable row level security;
